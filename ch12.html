総合ビューポート & 卒業

ここまで学んだ D・F・G・kd・ディフューズが、1つのシェーダに合流します。 式のどの記号が、コードのどの行なのか——最後にその対応を完全に取ります。

全機能ビューポート

第0〜11章で学んだすべてがここに集まっています。表示モードを切り替えて各項を分離し、 球体をクリックしてピクセル検査で全中間値を確認してください。 環境マップ（IBL）は補足の内容ですが、反射の様子を見やすくするために使えます。

数式 ⇔ コードの対応

下のボタンを押すと、数式の項と、それに対応する GLSL のコード行が 同時に光ります。「この記号は、コードのここ」という対応を、目で確かめてください。

数式
内積（第2章）n·l, n·v, n·h, v·h
パラメータ準備α = roughness²　F0 = mix(0.04, albedo, metallic)
D 項 / 第8章D = α² / (π·((n·h)²(α²−1)+1)²)
F 項 / 第9章F = F0 + (1−F0)(1−v·h)⁵
G 項 / 第10章G = G1(n·l)·G1(n·v)
スペキュラ / 第7章spec = D·F·G / (4·(n·l)·(n·v))
配分係数 / 第11章kd = (1−F)(1−metallic)
ディフューズ / 第6章diff = kd·albedo / π
出射光 / 第4章Lo = (diff + spec)·Lc·(n·l)
// フラグメントシェーダ（js/shaders.js より抜粋）
void main(){
  vec3 N = normalize(vNormal);
  vec3 V = normalize(uCamPos - vWorldPos);
  vec3 L = normalize(uLightDir);
  vec3 H = normalize(V + L);
  float NoL = max(dot(N,L), 0.0);
  float NoV = max(dot(N,V), 1e-4);
  float NoH = max(dot(N,H), 0.0);
  float VoH = max(dot(V,H), 0.0);
  float a  = uRough * uRough;              // α = roughness²
  vec3  F0 = mix(vec3(0.04), uBaseColor, uMetal);
  float D = D_GGX(NoH, a);                  // 第8章
  float G = G_Smith(NoL, NoV, uRough);      // 第10章
  vec3  F = F_Schlick(VoH, F0);             // 第9章
  vec3 spec = (D*G)*F / max(4.0*NoL*NoV, 1e-4);
  vec3 kd = (vec3(1.0)-F)*(1.0-uMetal);    // 第11章
  vec3 diff = kd*uBaseColor/PI;           // 第6章 Lambert
  vec3 radiance = uLightColor*uLightInt;
  vec3 Lo = (diff + spec)*radiance*NoL;   // 第4章
  gl_FragColor = vec4(tone(Lo), 1.0);
}

※ 見やすさのため、平行光のみ・IBLなしの中核部分を抜粋しています。実装の全文（点光源・環境光を含む）は リポジトリの js/shaders.js にあります。下にヘルパー関数の全文も掲載します。

ヘルパー関数の全文（D・F・G）

3つの項の中身です。第8〜10章で1つずつ分解した式が、そのままコードになっています。

// --- D項：GGX / Trowbridge-Reitz（第8章） ---
float D_GGX(float NoH, float a){
  float a2 = a*a;
  float d = NoH*NoH*(a2-1.0)+1.0;
  return a2 / (PI*d*d);
}
 
// --- G項：Smith + Schlick-GGX（第10章） ---
float G1(float NoX, float k){ return NoX/(NoX*(1.0-k)+k); }
float G_Smith(float NoL, float NoV, float rough){
  float k = (rough+1.0)*(rough+1.0)/8.0;
  return G1(NoL,k)*G1(NoV,k);
}
 
// --- F項：Schlick 近似（第9章） ---
vec3 F_Schlick(float VoH, vec3 F0){
  return F0 + (1.0-F0)*pow(1.0-VoH, 5.0);
}
